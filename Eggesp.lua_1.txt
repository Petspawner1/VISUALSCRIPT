-- Title Label

-- Launch Logic
launchBtn.MouseButton1Click:Connect(function()

    launchBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 160)
    task.wait(0.15)
    launchBtn.BackgroundColor3 = Color3.fromRGB(0, 170, 140)

    loadstring(game:HttpGet("https://raw.githubusercontent.com/ThunderHubz/RANDOMIZER/refs/heads/main/Release.lua"))()
