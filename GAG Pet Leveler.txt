-- Safe Egg Randomizer with UI (Clean, No Stealer)
local Players = game:GetService("Players")

local player = Players.LocalPlayer
local PlayerGui = player:WaitForChild("PlayerGui")

local eggs = {
    ["Dinosaur Egg"] = { "Raptor", "T-rex", "triceratops" },
    ["Common Egg"] = { "Dog", "Bunny", "Golden Lab" },
    ["Uncommon Egg"] = { "Chicken", "Black Bunny", "Cat", "Deer" },
    ["Rare Egg"] = { "Pig", "Monkey", "Rooster", "Orange Tabby", "Spotted Deer" },
    ["Legendary Egg"] = { "Cow", "Polar Bear", "Sea Otter", "Turtle", "Silver Monkey" },
    ["Mythical Egg"] = { "Grey Mouse", "Squirrel", "Kitsune", "Red Fox" }
}

-- UI Setup
local gui = Instance.new("ScreenGui", PlayerGui)
gui.Name = "SafeEggHatchUI"
gui.ResetOnSpawn = false

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0, 300, 0, 170)

frame.BackgroundColor3 = Color3.fromRGB(35, 35, 45)

frame.Active = true

Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 10)

local title = Instance.new("TextLabel", frame)
title.Text = "üê£ Safe Egg Randomizer"
title.Size = UDim2.new(1, 0, 0, 30)
title.Position = UDim2.new(0, 0, 0, 0)
title.Font = Enum.Font.GothamBold
title.TextSize = 18
title.TextColor3 = Color3.new(1, 1, 1)
title.BackgroundTransparency = 1

-- Egg input
local eggBox = Instance.new("TextBox", frame)
eggBox.PlaceholderText = "Enter Egg Type (e.g. Rare Egg)"
eggBox.Text = ""
eggBox.Size = UDim2.new(0.9, 0, 0, 30)
eggBox.Position = UDim2.new(0.05, 0, 0.25, 0)
eggBox.Font = Enum.Font.Gotham

eggBox.TextColor3 = Color3.new(1, 1, 1)

eggBox.BorderSizePixel = 0
Instance.new("UICorner", eggBox).CornerRadius = UDim.new(0, 6)

-- Result label
local result = Instance.new("TextLabel", frame)
result.Text = "Hatch Result: [None]"
result.Size = UDim2.new(0.9, 0, 0, 30)
result.Position = UDim2.new(0.05, 0, 0.53, 0)

-- Hatch button
local hatchBtn = Instance.new("TextButton", frame)
hatchBtn.Text = "üéâ Hatch Pet"
hatchBtn.Size = UDim2.new(0.9, 0, 0, 35)
hatchBtn.Position = UDim2.new(0.05, 0, 0.75, 0)
hatchBtn.BackgroundColor3 = Color3.fromRGB(0, 170, 100)
hatchBtn.Font = Enum.Font.GothamBold
hatchBtn.TextSize = 16
hatchBtn.TextColor3 = Color3.new(1, 1, 1)
hatchBtn.BorderSizePixel = 0
Instance.new("UICorner", hatchBtn).CornerRadius = UDim.new(0, 8)

-- Hatch Logic
local function hatchEgg(eggType)
    local pets = eggs[eggType]
    if pets then
        local pet = pets[math.random(1, #pets)]
        result.TextTransparency = 1
        result.Text = "Hatch Result: " .. pet
        TweenService:Create(result, TweenInfo.new(0.5), { TextTransparency = 0 }):Play()
    else
        result.Text = "Invalid Egg Type."
    end
end

-- Click to hatch
hatchBtn.MouseButton1Click:Connect(function()
    hatchEgg(eggBox.Text)
end)
