-- Logic to change age (stub logic, replace with your real method)
button.MouseButton1Click:Connect(function()
	local pet = player.Character and player.Character:FindFirstChild("EquippedPet")
	if pet then
		local age = pet:FindFirstChild("Age")
		if age and age:IsA("IntValue") then
			age.Value = 50
			petInfo.Text = "Equipped Pet: " .. pet.Name .. " (Age set to 50)"
		else
			petInfo.Text = "Pet age attribute not found."
		end
	else
		petInfo.Text = "No pet equipped."
	end
end)
